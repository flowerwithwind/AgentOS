### 1.3 开发环境搭建

**后端（Python 3.10+）**

> **Conda 环境说明**：项目可直接使用现有 conda 环境 `software`（Python 3.11.9），无需新建环境。该环境已安装 tornado 6.5.6、openai 2.40.0、requests 2.34.2 等 9/10 个项目依赖，仅需补装 `pymysql`。如使用其他 Python 环境（版本需 >= 3.10），按下方步骤创建虚拟环境即可。

**方式一：使用 conda software 环境（推荐）**

```bash
# 1. 激活 conda 环境
conda activate software

# 2. 补装缺失依赖（当前仅缺 pymysql）
pip install pymysql

# 3. 验证所有依赖已就绪
pip install -r requirements.txt   # 已安装的包会自动跳过

# 4. 配置环境变量（开发环境可选）
$env:XHAGENT_ENV="development"
$env:XHAGENT_COOKIE_SECRET="dev-secret-change-in-production"
$env:XHAGENT_DB_TYPE="sqlite"

# 5. 启动服务
python app.py
# 服务运行在 http://127.0.0.1:35001
```

**标准组件模板：**

```typescript
// components/common/StatCard.tsx
import React, { ErrorInfo, Component, ReactNode } from 'react';
import { Card, Statistic } from 'antd';
import type { CardProps } from 'antd';

/** StatCard 组件属性 */
interface StatCardProps {
  /** 卡片标题 */
  title: string;
  /** 数值 */
  value: number;
  /** 数值前缀图标 */
  prefix?: ReactNode;
  /** 数值后缀单位 */
  suffix?: string;
  /** 点击回调 */
  onClick?: () => void;
  /** 是否显示加载骨架 */
  loading?: boolean;
}

/** 错误边界：捕获子组件渲染错误 */
interface ErrorBoundaryState {
  hasError: boolean;
}

class ErrorBoundary extends Component<
  { children: ReactNode; fallback?: ReactNode },
  ErrorBoundaryState
> {
  state: ErrorBoundaryState = { hasError: false };

  static getDerivedStateFromError(): ErrorBoundaryState {
    return { hasError: true };
  }

  componentDidCatch(error: Error, info: ErrorInfo) {
    console.error('[StatCard] Render error:', error, info);
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback ?? <Card>组件渲染异常</Card>;
    }
    return this.props.children;
  }
}

/**
 * 统计卡片组件
 * 用于仪表盘等页面的数据概览卡片
 */
const StatCard: React.FC<StatCardProps> = ({
  title,
  value,
  prefix,
  suffix,
  onClick,
  loading = false,
}) => {
  return (
    <ErrorBoundary>
      <Card
        hoverable={!!onClick}
        onClick={onClick}
        style={{ borderRadius: 12, cursor: onClick ? 'pointer' : 'default' }}
      >
        <Statistic
          title={title}
          value={value}
          prefix={prefix}
          suffix={suffix}
          loading={loading}
        />
      </Card>
    </ErrorBoundary>
  );
};

export default React.memo(StatCard);
```

### 2.3 API 请求层规范

```typescript
// services/api.ts — 统一 Axios 实例与拦截器
import axios from 'axios';
import { message } from 'antd';

const API_BASE_URL = import.meta.env.VITE_API_BASE_URL || '/api';

const apiClient = axios.create({
  baseURL: API_BASE_URL,
  timeout: 30000,
  headers: { 'Content-Type': 'application/json' },
  withCredentials: true,
});

// 请求拦截器：注入认证信息
apiClient.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('xhagentos_token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error),
);

// 响应拦截器：统一错误处理
apiClient.interceptors.response.use(
  (response) => {
    const { code, message: msg, data } = response.data;
    if (code === 200 || code === 0) {
      return data;
    }
    message.error(msg || '请求失败');
    return Promise.reject(new Error(msg));
  },
  (error) => {
    if (error.response) {
      switch (error.response.status) {
        case 401:
          localStorage.removeItem('xhagentos_token');
          window.location.href = '/login';
          break;
        case 403:
          message.error('无权限访问');
          break;
        case 404:
          message.error('请求的资源不存在');
          break;
        case 429:
          message.error('请求过于频繁，请稍后重试');
          break;
        default:
          message.error(`服务器错误 (${error.response.status})`);
      }
    } else {
      message.error('网络连接异常');
    }
    return Promise.reject(error);
  },
);

export default apiClient;
```

```typescript
// hooks/useApi.ts — 统一请求状态管理 Hook
import { useState, useEffect, useCallback } from 'react';

interface UseApiResult<T> {
  data: T | null;
  loading: boolean;
  error: Error | null;
  refresh: () => void;
}

function useApi<T>(
  fetcher: () => Promise<T>,
  deps: unknown[] = [],
): UseApiResult<T> {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  const execute = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const result = await fetcher();
      setData(result);
    } catch (err) {
      setError(err instanceof Error ? err : new Error(String(err)));
    } finally {
      setLoading(false);
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, deps);

  useEffect(() => {
    execute();
  }, [execute]);

  return { data, loading, error, refresh: execute };
}

export default useApi;
```

### 2.4 状态管理规范

使用 Zustand 按模块拆分 store，仅存放跨组件共享的全局状态。

```typescript
// stores/useAuthStore.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface User {
  id: number;
  username: string;
  roleId: number;
  roleName: string;
}

interface AuthState {
  user: User | null;
  permissions: string[];
  setUser: (user: User, permissions: string[]) => void;
  logout: () => void;
  hasPermission: (code: string) => boolean;
}

const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      permissions: [],
      setUser: (user, permissions) => set({ user, permissions }),
      logout: () => set({ user: null, permissions: [] }),
      hasPermission: (code) => {
        const { user, permissions } = get();
        if (!user) return false;
        if (user.roleId === 1) return true; // 超级管理员
        return permissions.includes(code);
      },
    }),
    { name: 'xhagentos_auth' },
  ),
);

export default useAuthStore;
```

```typescript
// stores/useChatStore.ts
import { create } from 'zustand';

interface Message {
  id: number;
  role: 'user' | 'assistant' | 'system';
  content: string;
  createdAt: string;
  streaming?: boolean;
}

interface ChatState {
  currentSessionId: number | null;
  messages: Message[];
  isStreaming: boolean;
  setCurrentSession: (id: number) => void;
  addMessage: (msg: Message) => void;
  appendToLastMessage: (chunk: string) => void;
  setStreaming: (v: boolean) => void;
  clearMessages: () => void;
}

const useChatStore = create<ChatState>((set) => ({
  currentSessionId: null,
  messages: [],
  isStreaming: false,
  setCurrentSession: (id) => set({ currentSessionId: id, messages: [] }),
  addMessage: (msg) =>
    set((state) => ({ messages: [...state.messages, msg] })),
  appendToLastMessage: (chunk) =>
    set((state) => {
      const msgs = [...state.messages];
      if (msgs.length > 0) {
        const last = { ...msgs[msgs.length - 1] };
        last.content += chunk;
        msgs[msgs.length - 1] = last;
      }
      return { messages: msgs };
    }),
  setStreaming: (v) => set({ isStreaming: v }),
  clearMessages: () => set({ messages: [] }),
}));

xport default useChatStore;
```

```typescript
// hooks/useSSE.ts — 流式请求 Hook
async function* streamChat(
  url: string,
  body: Record<string, unknown>,
  signal: AbortSignal,
): AsyncGenerator<string> {
  const response = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
    signal,
    credentials: 'include',
  });

  if (!response.ok) {
    throw new Error(`HTTP ${response.status}`);
  }

  const reader = response.body!.getReader();
  const decoder = new TextDecoder();
  let buffer = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    buffer += decoder.decode(value, { stream: true });
    const lines = buffer.split('\n');
    buffer = lines.pop() || '';

    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const payload = line.slice(6);
        if (payload === '[DONE]') return;
        try {
          const parsed = JSON.parse(payload);
          if (parsed.content) yield parsed.content;
        } catch {
          // 跳过非 JSON 行
        }
      }
    }
  }
}
```

### 3.3 统一响应格式

所有 API 接口返回以下 JSON 结构：

```python
# utils/response.py
import time
from typing import Any, Optional

def success(data: Any = None, message: str = "success") -> dict:
    return {
        "code": 200,
        "message": message,
        "data": data,
        "timestamp": int(time.time()),
    }

def error(code: int, message: str, data: Any = None) -> dict:
    return {
        "code": code,
        "message": message,
        "data": data,
        "timestamp": int(time.time()),
    }

def paginated(items: list, total: int, page: int, page_size: int) -> dict:
    return success(data={
        "items": items,
        "total": total,
        "page": page,
        "pageSize": page_size,
        "totalPages": (total + page_size - 1) // page_size,
    })
```

**错误码体系：**

| 错误码 | 含义 | 场景 |
|--------|------|------|
| 200 | 成功 | 所有正常响应 |
| 400 | 参数错误 | 缺少必填字段、格式不合法 |
| 401 | 未认证 | 未登录或 Token 过期 |
| 403 | 无权限 | 角色权限不足 |
| 404 | 资源不存在 | ID 对应的记录不存在 |
| 409 | 资源冲突 | 名称重复、并发写入 |
| 422 | 业务校验失败 | 密码强度不足、状态不允许操作 |
| 429 | 请求过于频繁 | 触发 Rate Limiting |
| 500 | 服务器内部错误 | 未捕获异常 |
| 502 | 上游服务不可用 | AI 模型 API 调用失败 |

```python
def require_permission(permission_code):
    """权限校验装饰器
    
    用法：
        class UserListHandler(BaseHandler):
            @tornado.web.authenticated
            @require_permission("user:view")
            def get(self):
                ...
    """
    def decorator(method):
        @functools.wraps(method)
        def wrapper(self, *args, **kwargs):
            user = self.get_current_user()
            if not user:
                self.set_status(401)
                self.write({"code": 401, "message": "未认证"})
                return

            # 超级管理员跳过权限检查
            from app.models.user import UserRepository
            user_info = UserRepository.get_user_by_username(user)
            if user_info and user_info["role_id"] == 1:
                return method(self, *args, **kwargs)

            # 查询角色权限
            role_id = user_info["role_id"] if user_info else 2
            perms = RoleRepository.get_permissions_for_role(role_id)
            if "*" in perms or permission_code in perms:
                return method(self, *args, **kwargs)

            self.set_status(403)
            self.write({"code": 403, "message": "无权限访问"})
        return wrapper
    return decorator
```

**Rate Limiting 方案：**

基于内存的滑动窗口限流，开发阶段用内存存储，生产环境可替换为 Redis。

```python
# middleware/rate_limit.py
import time
from collections import defaultdict

class RateLimiter:
    def __init__(self, max_requests=60, window_seconds=60):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.requests = defaultdict(list)

    def is_allowed(self, key: str) -> bool:
        now = time.time()
        window_start = now - self.window_seconds
        # 清理过期记录
        self.requests[key] = [t for t in self.requests[key] if t > window_start]
        if len(self.requests[key]) >= self.max_requests:
            return False
        self.requests[key].append(now)
        return True

# 全局实例
rate_limiter = RateLimiter()
```

限流策略：普通 API 60 次/分钟，AI 对话接口 10 次/分钟，登录接口 5 次/分钟。

### 3.6 性能开发规范

**数据库查询规范：**

- 禁止 N+1 查询：关联数据必须用 JOIN 一条 SQL 获取
- 禁止 `SELECT *`，必须明确列出字段名
- 复杂查询必须加索引，索引命名规则：`idx_表名_字段名`
- 分页查询必须使用 `LIMIT` + `OFFSET`
- 统计查询结果超过 1000 行时必须加分页

```python
# models/digital_employee.py — 修复后的 list 方法
@staticmethod
def list(page: int = 1, page_size: int = 10, keyword: str = ''):
    offset = (page - 1) * page_size
    with get_connection() as conn:
        # 单条 JOIN 查询获取员工 + 技能名称
        base_sql = """
            SELECT de.id, de.name, de.avatar_url, de.welcome_message,
                   de.system_prompt, de.status, de.create_at,
                   GROUP_CONCAT(s.name, ', ') AS skill_names
            FROM digital_employees de
            LEFT JOIN employee_skills es ON de.id = es.employee_id
            LEFT JOIN skills s ON es.skill_id = s.id
        """
        if keyword:
            rows = conn.execute(
                base_sql + " WHERE de.name LIKE ? GROUP BY de.id ORDER BY de.id DESC LIMIT ? OFFSET ?",
                (f'%{keyword}%', page_size, offset)
            ).fetchall()
            total = conn.execute(
                "SELECT COUNT(*) AS cnt FROM digital_employees WHERE name LIKE ?",
                (f'%{keyword}%',)
            ).fetchone()["cnt"]
        else:
            rows = conn.execute(
                base_sql + " GROUP BY de.id ORDER BY de.id DESC LIMIT ? OFFSET ?",
                (page_size, offset)
            ).fetchall()
            total = conn.execute(
                "SELECT COUNT(*) AS cnt FROM digital_employees"
            ).fetchone()["cnt"]
        return [dict(r) for r in rows], total
```

```python
# controllers/chat.py — SSE 流式响应 Handler
import tornado.web
import json
from app.controllers.base import BaseHandler
from app.services.chat_service import chat_completion

class ChatSendHandler(BaseHandler):
    @tornado.web.authenticated
    async def post(self):
        body = self.get_json()
        session_id = body.get("sessionId")
        content = body.get("content", "").strip()
        model_id = body.get("modelId")

        if not content:
            return self.write({"code": 400, "message": "消息不能为空"})

        # 设置 SSE 响应头
        self.set_header("Content-Type", "text/event-stream")
        self.set_header("Cache-Control", "no-cache")
        self.set_header("Connection", "keep-alive")
        self.set_header("X-Accel-Buffering", "no")

        full_reply = []
        try:
            async for chunk_text in chat_completion(model_id, self._build_messages(session_id, content)):
                full_reply.append(chunk_text)
                data = json.dumps({"content": chunk_text, "done": False}, ensure_ascii=False)
                self.write(f"data: {data}\n\n")
                self.flush()

            # 流结束标记
            reply = "".join(full_reply)
            self._save_message(session_id, content, reply)
            end_data = json.dumps({
                "sessionId": session_id,
                "tokenUsage": len(reply),
                "done": True
            }, ensure_ascii=False)
            self.write(f"data: {end_data}\n\n")
            self.write("data: [DONE]\n\n")
            self.flush()
        except Exception as e:
            error_data = json.dumps({"error": str(e)}, ensure_ascii=False)
            self.write(f"data: {error_data}\n\n")
            self.flush()

        self.finish()
```
